Angular - Routage & Services

10 Cours Théoriques Approfondis

1. RouterModule et AppRoutingModule pour centraliser toutes les routes de l'application.

2. Définir routes simples et paramétrées : { path: 'user/:id', component: UserComponent }.

3. Utiliser routerLink pour naviguer depuis les templates HTML.

4. <router-outlet> comme point d’injection dynamique des composants selon la route.

5. Récupérer paramètres de route : this.route.snapshot.paramMap.get('id').

6. Navigation programmatique avec this.router.navigate(['/home']).

7. Services Angular : centraliser logique métier et accès aux API.

8. HttpClientModule pour effectuer des appels REST vers backend.

9. Observables et subscribe() pour recevoir les données asynchrones.

10. Manipulation des données avec map() ou pipe() pour transformation avant affichage.

Exemples Commentés


/* 1. AppRoutingModule avec routes simples et paramétrées */
const routes: Routes = [
  { path: '', redirectTo: 'home', pathMatch: 'full' },
  { path: 'home', component: HomeComponent },
  { path: 'users', component: UsersComponent },
  { path: 'user/:id', component: UserComponent },
  { path: '**', component: PageNotFoundComponent }
];
@NgModule({
  imports: [RouterModule.forRoot(routes)],
  exports: [RouterModule]
})
export class AppRoutingModule {}

/* 2. Template avec routerLink */
<a routerLink="/users">Liste des utilisateurs</a>

/* 3. RouterOutlet pour injection dynamique */
<router-outlet></router-outlet>

/* 4. Service Angular pour API utilisateurs */
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class UserService {
  private apiUrl = 'http://localhost:5000/users';
  constructor(private http: HttpClient) {}
  getUsers(): Observable<any> { return this.http.get(this.apiUrl); }
  getUser(id: number): Observable<any> { return this.http.get(`${this.apiUrl}/${id}`); }
}

/* 5. Appel service et affichage */
this.userService.getUsers().subscribe(users => {
  this.users = users.map(u => ({ ...u, fullName: u.name.toUpperCase() }));
});

/* 6. Navigation programmatique */
goToUser(userId: number) {
  this.router.navigate(['/user', userId]);
}
